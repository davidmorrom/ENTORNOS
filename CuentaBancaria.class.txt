class CuentaBancaria
{

		this.numCuenta = numCuenta;

	public void ingreso(double cantidad)
	{
		if(cantidad > 0)
			saldo = saldo + cantidad;

	}

	{
		if(cantidad > 0)
		{
			if(cantidad > saldo)
				System.out.println("Saldo Insuficiente");
				
				else
					saldo = saldo - cantidad;
		}
		else

	}
